# 🏗️ CURSOR PROMPT — نظام إدارة شركة إنتاج متكامل

-----

## 🎯 نظرة عامة على المشروع

أنت مطلوب منك تبني نظام إدارة متكامل لشركة إنتاج (Production Company).
النظام يضم: إدارة ليدز، نظام محاسبي، إدارة معدات، جدول إنتاج، وتحليلات.

-----

## 🛠️ Tech Stack المطلوب

```
Frontend:     Next.js 14 (App Router) + TypeScript
Styling:      Tailwind CSS + shadcn/ui
Database:     Supabase (PostgreSQL + Auth + Storage + Realtime)
Automation:   n8n (self-hosted) — للـ webhooks والـ workflows
AI:           OpenAI API (gpt-4o-mini للملخصات، Whisper للمكالمات)
Ads APIs:     Meta Marketing API + Google Ads API (Leads)
Mobile App:   React Native (Expo) — Android only — لتسجيل المكالمات
Deployment:   Vercel (frontend) + Railway أو VPS (n8n)
Direction:    RTL — عربي بالكامل
```

## 🗄️ Database Schema (Supabase)

```sql
-- المستخدمين
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,
  full_name TEXT NOT NULL,
  role TEXT NOT NULL CHECK (role IN ('owner','accountant','sales_manager','sales_rep')),
  is_online BOOLEAN DEFAULT false,
  last_seen TIMESTAMPTZ,
  skills TEXT[], -- ['english','large_companies','real_estate']
  languages TEXT[], -- ['arabic','english','french']
  performance_score DECIMAL DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- الليدز
CREATE TABLE leads (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  source TEXT NOT NULL CHECK (source IN ('facebook','google','manual')),
  ad_name TEXT,
  campaign_id TEXT,
  full_name TEXT,
  phone TEXT,
  email TEXT,
  company_name TEXT,
  company_size TEXT CHECK (company_size IN ('small','medium','large','enterprise')),
  language TEXT DEFAULT 'arabic',
  message TEXT,
  score INTEGER DEFAULT 0, -- Lead Score من 0-100
  status TEXT DEFAULT 'new' CHECK (status IN ('new','assigned','contacted','interested','proposal','won','lost')),
  assigned_to UUID REFERENCES users(id),
  assigned_at TIMESTAMPTZ,
  first_contact_deadline TIMESTAMPTZ,
  first_contacted_at TIMESTAMPTZ,
  notes TEXT,
  lost_reason TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- سجل نقل الليدز
CREATE TABLE lead_transfers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  lead_id UUID REFERENCES leads(id),
  from_user UUID REFERENCES users(id),
  to_user UUID REFERENCES users(id),
  reason TEXT, -- 'timeout','manual','reassign'
  transferred_at TIMESTAMPTZ DEFAULT now()
);

-- المحادثات والتفاعلات
CREATE TABLE lead_interactions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  lead_id UUID REFERENCES leads(id),
  user_id UUID REFERENCES users(id),
  type TEXT CHECK (type IN ('call','whatsapp','email','note','meeting')),
  duration_seconds INTEGER,
  recording_url TEXT, -- رابط التسجيل في Supabase Storage
  transcript TEXT,    -- النص بعد Whisper
  ai_summary TEXT,    -- ملخص GPT
  sentiment TEXT CHECK (sentiment IN ('positive','neutral','negative')),
  next_action TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- الصفقات والمشاريع
CREATE TABLE deals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  lead_id UUID REFERENCES leads(id),
  title TEXT NOT NULL,
  value DECIMAL NOT NULL,
  status TEXT DEFAULT 'open' CHECK (status IN ('open','won','lost')),
  expected_close_date DATE,
  actual_close_date DATE,
  created_by UUID REFERENCES users(id),
  created_at TIMESTAMPTZ DEFAULT now()
);

-- الفواتير
CREATE TABLE invoices (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  deal_id UUID REFERENCES deals(id),
  invoice_number TEXT UNIQUE NOT NULL,
  type TEXT CHECK (type IN ('income','expense')),
  amount DECIMAL NOT NULL,
  tax_amount DECIMAL DEFAULT 0,
  status TEXT DEFAULT 'pending' CHECK (status IN ('pending','paid','overdue','cancelled')),
  due_date DATE,
  paid_date DATE,
  category TEXT, -- 'production','ads','salaries','equipment','other'
  description TEXT,
  file_url TEXT, -- PDF مرفق
  created_by UUID REFERENCES users(id),
  created_at TIMESTAMPTZ DEFAULT now()
);

-- المصروفات
CREATE TABLE expenses (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  category TEXT NOT NULL,
  amount DECIMAL NOT NULL,
  description TEXT,
  date DATE NOT NULL,
  receipt_url TEXT,
  created_by UUID REFERENCES users(id),
  created_at TIMESTAMPTZ DEFAULT now()
);

-- المعدات
CREATE TABLE equipment (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  category TEXT, -- 'camera','lighting','audio','lens','other'
  serial_number TEXT,
  status TEXT DEFAULT 'available' CHECK (status IN ('available','booked','maintenance','out')),
  purchase_date DATE,
  purchase_price DECIMAL,
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- حجز المعدات
CREATE TABLE equipment_bookings (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  equipment_id UUID REFERENCES equipment(id),
  production_session_id UUID REFERENCES production_sessions(id),
  booked_by UUID REFERENCES users(id),
  date DATE NOT NULL,
  start_time TIME NOT NULL,
  end_time TIME NOT NULL,
  returned_at TIMESTAMPTZ,
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- جلسات الإنتاج
CREATE TABLE production_sessions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  client_name TEXT,
  deal_id UUID REFERENCES deals(id),
  date DATE NOT NULL,
  start_time TIME NOT NULL,
  end_time TIME NOT NULL,
  location TEXT,
  team_members UUID[], -- array of user IDs
  status TEXT DEFAULT 'scheduled' CHECK (status IN ('scheduled','in_progress','completed','cancelled')),
  notes TEXT,
  created_by UUID REFERENCES users(id),
  created_at TIMESTAMPTZ DEFAULT now(),
  CONSTRAINT no_overlap UNIQUE(date, start_time) -- منع التعارض
);

-- إعدادات النظام
CREATE TABLE system_settings (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  key TEXT UNIQUE NOT NULL,
  value JSONB,
  updated_at TIMESTAMPTZ DEFAULT now()
);
-- مثال: lead_response_time_minutes = 30
-- مثال: lead_distribution_rules = {...}

-- إعدادات الإعلانات
CREATE TABLE ad_integrations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  platform TEXT CHECK (platform IN ('facebook','google')),
  access_token TEXT,
  account_id TEXT,
  page_id TEXT,
  form_ids TEXT[],
  is_active BOOLEAN DEFAULT true,
  last_sync TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- سجل نقاط الموظفين (Gamification)
CREATE TABLE performance_points (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id),
  points INTEGER NOT NULL,
  reason TEXT, -- 'lead_won','fast_response','lead_lost','late_response'
  created_at TIMESTAMPTZ DEFAULT now()
);
```

## ⚡ الـ Features التفصيلية

### Lead Scoring Algorithm

```typescript
// lib/lead-scoring.ts
function calculateLeadScore(lead: Lead): number {
  let score = 0;
  
  // المصدر
  if (lead.source === 'google') score += 20;
  if (lead.source === 'facebook') score += 15;
  
  // حجم الشركة
  if (lead.company_size === 'enterprise') score += 30;
  if (lead.company_size === 'large') score += 20;
  if (lead.company_size === 'medium') score += 10;
  
  // اللغة (إنجليزي = أجنبي = أعلى قيمة)
  if (lead.language === 'english') score += 15;
  
  // وقت التقديم (أوقات العمل أفضل)
  const hour = new Date(lead.created_at).getHours();
  if (hour >= 9 && hour <= 17) score += 10;
  
  // وجود رقم تليفون
  if (lead.phone) score += 10;
  
  // وجود اسم شركة
  if (lead.company_name) score += 5;
  
  return Math.min(score, 100);
}
```

### Lead Distribution Engine

```typescript
// lib/lead-distributor.ts
async function distributeLead(lead: Lead): Promise<string> {
  const rules = await getDistributionRules(); // من system_settings
  const onlineReps = await getOnlineReps();   // is_online = true
  
  // تطبيق القواعد بالترتيب
  for (const rule of rules) {
    if (rule.condition === 'language' && lead.language === rule.value) {
      const match = onlineReps.find(r => r.languages.includes(rule.value));
      if (match) return match.id;
    }
    if (rule.condition === 'company_size' && lead.company_size === rule.value) {
      // أعطيه لأعلى موظف performance_score
      const best = onlineReps.sort((a,b) => b.performance_score - a.performance_score)[0];
      if (best) return best.id;
    }
  }
  
  // لو مفيش rule — Round Robin
  return roundRobin(onlineReps);
}
```
